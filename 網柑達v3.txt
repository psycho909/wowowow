@use "sass:math";
@function strip-unit($number) {
	@if type-of($number) == "number" and not unitless($number) {
		@return math.div($number, $number * 0 + 1);
	}

	@return $number;
}
@function rem($size, $breakpoint: 768, $zoom: 1) {
	@return math.div(strip-unit($size), $breakpoint) * $zoom * 1rem;
}
